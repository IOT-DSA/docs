def type=(val_type)
    _type = nil
    @val_types.each do |t|
        if val_type == t
            _type = val_type
            break
        elsif t == 'enum' and val_type.match(/enum\[[^\]]+\]/) != nil
            _type = val_type
            break
        end
    end
    raise "'#{val_type}' is not a known Value type" if _type.nil?
    @type = _type
end
