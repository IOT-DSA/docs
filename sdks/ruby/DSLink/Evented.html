def fire_event(event, data)
    return unless @__events && @__events[event]
    @__events[event].each do |evt|
        evt[:cb].call(data)
    end
end

def on(event, id, &block)
    @__events ||= {}
    @__events[event] ||= []
    @__events[event] << { id: id, cb: block }
end
