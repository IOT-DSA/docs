Stream<String> discover({Duration timeout: const Duration(seconds: 5)}) {
  _send("DISCOVER", "239.255.255.230", 1900);
  return _brokerController.stream
      .timeout(timeout, onTimeout: (EventSink sink) {
    sink.close();
  });
}
