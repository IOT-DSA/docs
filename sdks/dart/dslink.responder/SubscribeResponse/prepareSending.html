void prepareSending() {
  if (_sendingAfterAck) {
    return;
  }
  if (_waitingAckCount > ConnectionProcessor.ACK_WAIT_COUNT) {
    _sendingAfterAck = true;
    return;
  }
  if (!_pendingSending) {
    _pendingSending = true;
    responder.addProcessor(this);
  }
}
