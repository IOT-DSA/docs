void addTraceCallback(ResponseTraceCallback _traceCallback) {
  subscriptions.forEach((path, controller) {
    ResponseTrace update = new ResponseTrace(
        controller.node.path, 'subscribe', 0, '+');
    _traceCallback(update);
  });
}
