void unsubscribe(Map m) {
  if (m['sids'] is List) {
    int rid = m['rid'];
    for (Object sid in m['sids']) {
      if (sid is int) {
        _subscription.remove(sid);
      }
    }
    _closeResponse(m['rid']);
  } else {
    _closeResponse(m['rid'], error: DSError.INVALID_PATHS);
  }
}
