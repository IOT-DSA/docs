/// Called by the link internals to remove a config from this node.
Response removeConfig(String name, Responder responder, Response response) {
  if (response != null) {
    return response..close();
  } else {
    if (!name.startsWith(r"$")) {
      name = "\$${name}";
    }
    configs.remove(name);
    return null;
  }
}
