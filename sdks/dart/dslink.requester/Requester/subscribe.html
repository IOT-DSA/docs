ReqSubscribeListener subscribe(String path, callback(ValueUpdate update),
    [int qos = 0]) {
  RemoteNode node = nodeCache.getRemoteNode(path);
  node._subscribe(this, callback, qos);
  return new ReqSubscribeListener(this, path, callback);
}
