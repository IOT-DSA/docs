ReqSubscribeListener subscribe(String path, callback(ValueUpdate),
    [int cacheLevel = 1]) {
  RemoteNode node = nodeCache.getRemoteNode(path);
  node._subscribe(this, callback, cacheLevel);
  return new ReqSubscribeListener(this, path, callback);
}
