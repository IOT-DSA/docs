void onProfileUpdated() {
  if (_ready) {
    if (request.streamStatus != StreamStatus.initialize) {
      _controller.add(new RequesterListUpdate(
          node, changes.toList(), request.streamStatus));
      changes.clear();
    }
    if (request.streamStatus == StreamStatus.closed) {
      _controller.close();
    }
  }
}
