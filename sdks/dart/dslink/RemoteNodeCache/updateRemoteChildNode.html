RemoteNode updateRemoteChildNode(RemoteNode parent, String name, Map m) {
  String path;
  if (parent.remotePath == '/') {
    path = '/$name';
  } else {
    path = '${parent.remotePath}/$name';
  }
  RemoteNode rslt;
  if (_nodes.containsKey(path)) {
    rslt = _nodes[path];
    rslt.updateRemoteChildData(m, this);
  } else {
    rslt = new RemoteNode(path);
    _nodes[path] = rslt;
    rslt.updateRemoteChildData(m, this);
  }
  return rslt;
}
