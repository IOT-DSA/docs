/// Updates table rows to [rows].
/// [stat] is the stream status.
/// [meta] is the action result metadata.
void update(List rows, [String stat, Map meta]) {
  if (this.rows == null) {
    this.rows = rows;
  } else {
    this.rows.addAll(rows);
  }
  this.meta = meta;
  if (stat != null) {
    status = stat;
  }
  write();
}
