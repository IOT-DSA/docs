void forEachAttribute(void callback(String name, Object value)) {
  attributes.forEach(callback);
  if (profile != null) {
    profile.attributes.forEach((String str, Object val) {
      if (!attributes.containsKey(str)) {
        callback(str, val);
      }
    });
  }
}
