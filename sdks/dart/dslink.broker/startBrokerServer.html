Future<DsHttpServer> startBrokerServer(int port, {bool persist: true}) async {
  var broker = new BrokerNodeProvider();
  broker.shouldSaveFiles = persist;
  var server = new DsHttpServer.start(
      "0.0.0.0",
      httpPort: port,
      linkManager: broker,
      nodeProvider: broker
  );
  await server.onServerReady;
  return server;
}
