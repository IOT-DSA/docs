void onPingTimer(Timer t) {
  if (_dataReceiveCount >= 3) {
    this._onDone();
    return;
  }
  _dataReceiveCount++;

  if (_dataSent) {
    _dataSent = false;
    return;
  }
  addConnCommand(null, null);
}
