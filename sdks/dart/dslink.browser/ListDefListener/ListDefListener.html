ListDefListener(this.node, this.requester,
    void callback(RequesterListUpdate)) {
  listener = requester.list(node.remotePath).listen((RequesterListUpdate update) {
    ready = update.streamStatus != StreamStatus.initialize;
    callback(update);
  });
}
