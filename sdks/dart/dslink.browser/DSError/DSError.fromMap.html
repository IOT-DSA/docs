DSError.fromMap(Map m) {
  if (m["type"] is String) {
    type = m["type"];
  }
  if (m["msg"] is String) {
    msg = m["msg"];
  }
  if (m["path"] is String) {
    path = m["path"];
  }
  if (m["phase"] is String) {
    phase = m["phase"];
  }
  if (m["detail"] is String) {
    detail = m["detail"];
  }
}
