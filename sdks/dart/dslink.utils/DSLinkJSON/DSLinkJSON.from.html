factory DSLinkJSON.from(Map<String, dynamic> map) {
  var j = new DSLinkJSON();
  j._json = map;
  j.name = map["name"];
  j.version = map["version"];
  j.description = map["description"];
  j.main = map["main"];
  j.engines = map["engines"];
  j.configs = map["configs"];
  j.getDependencies = map["getDependencies"];
  return j;
}
